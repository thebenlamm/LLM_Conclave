```typescript
async function handleDiscuss(args: { /* keep same signature */ }, server: Server) {
  const runner = new DiscussionRunner();

  // Build progress callback that forwards to MCP logging
  const onProgress = (event: { type: string; message: string }) => {
    const level = event.type === 'error' ? 'warning' : 'info';
    server.sendLoggingMessage({
      level,
      logger: 'llm-conclave',
      data: event.message,
    }).catch((err: any) => { console.error('[MCP] Log send failed:', err?.message); });
  };

  const { result, sessionId, logFilePath, timedOut, effectiveTimeout } = await runner.run({
    task: args.task,
    config: args.config,
    projectPath: args.project,
    personas: args.personas,
    rounds: args.rounds ?? 4,
    minRounds: args.min_rounds ?? 2,
    dynamic: args.dynamic ?? false,
    selectorModel: args.selector_model ?? DEFAULT_SELECTOR_MODEL,
    judgeModel: args.judge_model,
    judgeInstructions: args.judge_instructions,
    timeout: args.timeout ?? 0,
    contextOptimization: args.context_optimization ?? false,
    onProgress,
    validateProjectPath: (p) => validatePath(p, process.cwd()),
  });

  // Format output (keep existing formatting logic — it's handler-specific)
  const format = args.format ?? 'markdown';
  let outputText: string;

  if (format === 'json' || format === 'both') {
    const jsonResult = formatDiscussionResultJson(result, logFilePath, sessionId);
    if (format === 'both') {
      let markdown = '';
      if (result.degraded) markdown += `**Discussion aborted:** ${result.degradedReason}\n\n`;
      if (result.timedOut) markdown += `**Discussion timed out after ${effectiveTimeout}s** (${result.rounds} rounds completed)\n\n`;
      markdown += formatDiscussionResult(result, logFilePath, sessionId, { includeTranscript: !!result.timedOut });
      outputText = JSON.stringify({ ...jsonResult, markdown_summary: markdown }, null, 2);
    } else {
      outputText = JSON.stringify(jsonResult, null, 2);
    }
  } else {
    let summary = '';
    if (result.degraded) summary += `**Discussion aborted:** ${result.degradedReason}\nCheck MCP server logs at ~/.llm-conclave/mcp-server.log\n\n`;
    if (result.timedOut) summary += `**Discussion timed out after ${effectiveTimeout}s** (${result.rounds} rounds completed)\n\n`;
    summary += formatDiscussionResult(result, logFilePath, sessionId, { includeTranscript: !!result.timedOut });
    outputText = summary;
  }

  return { content: [{ type: 'text', text: outputText }] };
}
```

```typescript
async function handleContinue(args: { session_id?: string; task: string; reset?: boolean }, server: Server) {
  const { session_id, task, reset = false } = args;
  const sessionManager = new SessionManager();
  const continuationHandler = new ContinuationHandler();

  // Session loading (continuation-specific, keep as-is)
  let session;
  if (session_id) {
    session = await sessionManager.loadSession(session_id);
    if (!session) throw new Error(`Session '${session_id}' not found. Use llm_conclave_sessions to list available sessions.`);
  } else {
    session = await sessionManager.getMostRecentSession();
    if (!session) throw new Error('No sessions found. Run a discussion first using llm_conclave_discuss.');
  }

  // Validation (continuation-specific, keep as-is)
  const validation = continuationHandler.validateResumable(session);
  if (!validation.isValid) throw new Error(`Cannot resume session: ${validation.warnings.join(', ')}`);

  // Prepare continuation context (continuation-specific, keep as-is)
  const prepared = continuationHandler.prepareForContinuation(session, task, {
    resetDiscussion: reset,
    includeFullHistory: !reset,
  });

  // Fix legacy model names (continuation-specific helper)
  const fixLegacyModel = (model: string | undefined, fallback: string): string => {
    if (!model) return fallback;
    if (model.endsWith('Provider')) return fallback;
    return model;
  };

  // Rebuild config from session (continuation-specific)
  const config: any = {
    max_rounds: session.maxRounds || 4,
    min_rounds: session.minRounds ?? 0,
    agents: {},
    judge: {
      model: fixLegacyModel(session.judge?.model, 'gpt-4o'),
      prompt: session.judge?.systemPrompt || 'You are a judge evaluating agent responses.',
    },
  };
  for (const agent of session.agents) {
    config.agents[agent.name] = {
      model: fixLegacyModel(agent.model, 'gpt-4o'),
      prompt: agent.systemPrompt,
    };
  }

  // Delegate to DiscussionRunner with priorHistory
  const runner = new DiscussionRunner();
  const onProgress = (event: { type: string; message: string }) => {
    server.sendLoggingMessage({
      level: event.type === 'error' ? 'warning' : 'info',
      logger: 'llm-conclave',
      data: event.message,
    }).catch((err: any) => { console.error('[MCP] Log send failed:', err?.message); });
  };

  const { result, sessionId: newSessionId, logFilePath } = await runner.run({
    task: prepared.newTask,
    rounds: config.max_rounds,
    minRounds: config.min_rounds,
    onProgress,
    priorHistory: prepared.mergedHistory.map((msg: any) => ({
      role: msg.role,
      content: msg.content,
      speaker: msg.speaker || (msg.role === 'user' ? 'System' : 'Assistant'),
    })),
    // Pass pre-built config directly — don't re-resolve from file
    // DiscussionRunner needs to accept a resolvedConfig option for this case
  });

  // Link to parent session (continuation-specific)
  // Note: DiscussionRunner saves a session, but we need to update it with parentSessionId
  // This may require a post-save update via sessionManager, or DiscussionRunner accepts parentSessionId option

  // Format response (continuation-specific format)
  let output = `# Continuation of Session ${session.id}\n\n`;
  output += `**Original Task:** ${session.task}\n\n`;
  output += `**Follow-up:** ${task}\n\n`;
  output += formatDiscussionResult(result, logFilePath, newSessionId);

  return { content: [{ type: 'text', text: output }] };
}
```

**IMPORTANT:** handleContinue rebuilds config from the session, not from ConfigCascade.resolve(). DiscussionRunner.run() currently resolves config from a path. resolvedConfig?: any;  // Pre-resolved config (for continuation, bypasses ConfigCascade)
In run(), use `resolvedConfig ?? ConfigCascade.resolve({ config: options.config })`.

Also add `parentSessionId?: string` to DiscussionRunnerOptions so the saved session can be linked to its parent.
